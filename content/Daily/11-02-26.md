const issPeriod = 92.68; // minutes
const issAltitude = 420; // km

// TLE-derived mean motion: ~15.5 orbits/day
const orbitsPerDay = 1440 / issPeriod;

// Simplified ground track calculation
// ISS longitude shifts ~22.5° west each orbit due to Earth's rotation
const rad = (d) => d * Math.PI / 180;
const deg = (r) => r * 180 / Math.PI;

// Minutes since arbitrary epoch (we just need relative position)
const minsToday = now.getUTCHours() * 60 + now.getUTCMinutes();
const dayOfYear = Math.floor((now - new Date(now.getFullYear(), 0, 0)) / 86400000);

// Approximate current orbital position
const orbitFraction = (minsToday / issPeriod) % 1;
const issLat = issInclination * Math.sin(orbitFraction * 2 * Math.PI);

// Longitude drifts ~360° per day relative to ground
const issLon = ((minsToday / 1440) * -360 + (dayOfYear * 15.5 * orbitsPerDay)) % 360;
const issLonNorm = issLon > 180 ? issLon - 360 : issLon < -180 ? issLon + 360 : issLon;

// Distance from Cardiff
const dLat = issLat - latitude;
const dLon = issLonNorm - longitude;
const approxDist = Math.sqrt(dLat*dLat + dLon*dLon);

// Rough visibility check (ISS visible when <~2000km ground distance and sun angle right)
const isClose = approxDist < 30; // Within ~30° great circle

// Where is it roughly?
const latDir = issLat >= 0 ? 'N' : 'S';
const lonDir = issLonNorm >= 0 ? 'E' : 'W';

           issLonNorm >= -130 && issLonNorm < -60 ? 'over North America' :
           'over the northern hemisphere';
} else if (issLat < -30) {
